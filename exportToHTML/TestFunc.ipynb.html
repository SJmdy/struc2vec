180      "def alias_setup(probs):\n", 
181      "    '''\n", 
182      "\n", 
183      "    :param probs: 某个概率分布\n", 
184      "    :return: Alias数组与Prob数组\n", 
185      "    '''\n", 
186      "    K =len(probs) # K为类别数目\n", 
187      "    Prob =np.zeros(K) # 对应Prob数组：落在原类型的概率\n", 
188      "    Alias =np.zeros(K,dtype=np.int) # 对应Alias数组：每一列第二层的类型\n", 
189      "\n", 
190      "    #Sort the data into the outcomes with probabilities\n", 
191      "    #that are larger and smaller than 1/K\n", 
192      "    smaller =[] # 存储比1小的列\n", 
193      "    larger =[] # 存储比1大的列\n", 
194      "\n", 
195      "    for kk,prob in enumerate(probs):\n", 
196      "        Prob[kk] =K*prob # 概率（每个类别概率乘以K，使得总和为K）\n", 
197      "        if Prob[kk] <1.0: # 然后分为两类：大于1的和小于1的\n", 
198      "            smaller.append(kk)\n", 
199      "        else:\n", 
200      "            larger.append(kk)\n", 
201      "\n", 
202      "    # Loop though and create little binary mixtures that appropriately allocate\n", 
203      "    # the larger outcomes over the overall uniform mixture.\n", 
204      "\n", 
205      "    #通过拼凑，将各个类别都凑为1\n", 
206      "    while len(smaller) > 0 and len(larger) > 0:\n", 
207      "        small = smaller.pop()\n", 
208      "        large = larger.pop()\n", 
209      "\n", 
210      "        Alias[small] = large #填充Alias数组\n", 
211      "        Prob[large] = Prob[large]-(1.0 - Prob[small]) #将大的分到小的上\n", 
212      "\n", 
213      "        if Prob[large] <1.0:\n", 
214      "            smaller.append(large)\n", 
215      "        else:\n", 
216      "            larger.append(large)\n",
