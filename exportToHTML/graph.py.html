31       def normalize_weight(self):
32           for node, node_neighboors in self.G.adj.items():
33               sum_weights = sum([node_neighboors[i]['weight'] for i in node_neighboors])
34               for nbr in node_neighboors:
35                   node_neighboors[nbr]['weight'] /= sum_weights
36           return self.G
