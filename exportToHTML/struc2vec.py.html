48       def get_fk_uv(self):
49           """ 
50               s_nbrs: { 
51                   k: { 
52                       node: the degree sequence of node 
53                   } 
54               } 
55           """
56           for k in range(0, self.diameter):
57               for u in self.nodes:
58                   for v in self.nodes:
59                       # 只有当u、v都有第k跳的节点时，fk_uv[u][v]才有意义
60                       if len(self.s_nbrs[k][u]) > 0 and len(self.s_nbrs[k][v]) > 0:
61                           if u != v:
62                               u_nbrs = self.s_nbrs[k][u]
63                               v_nbrs = self.s_nbrs[k][v]
64                               dist_uv = self.G.get_dtw_distance(u_nbrs, v_nbrs)
65                               # print(k, u, v)
66                               self.fk_uv[k][u][v] = self.fk_uv[k - 1][u][v] + dist_uv if k > 0 else dist_uv
67                           else:
68                               self.fk_uv[k][u][v] = 1
69   
70       def get_wk_uv(self):
71           self.wk_uv = np.exp(-1 * self.fk_uv)
72   
73       def get_wk_average(self):
74           for k in range(0, self.diameter):
75               edges_number = (self.edges_number * (self.edges_number - 1)) / 2 if k > 0 else self.edges_number
76               # print(self.wk_uv[k].sum())
77               # print(self.wk_uv[k])
78               self.wk_average[k] = self.wk_uv[k].sum() / edges_number

80       def get_tku(self):
81           for k in range(0, self.diameter):
82               for u in self.nodes:
83                   self.tku[k][u] = np.size(self.wk_uv[k][u][self.wk_uv[k][u] > self.wk_average[k]])
84   
85       def get_wku(self):
86           for k in range(0, self.diameter):
87               for u in self.nodes:
88                   # wku[k][u][0]: w(uk, uk+1)
89                   # wku[k][u][1]: w(uk, uk-1)
90                   self.wku[k][u][0] = np.log10(self.tku[k][u] + np.e)
91                   self.wku[k][u][1] = 1
92   
93       def get_zku(self):
94           for k in range(0, self.diameter):
95               for u in self.nodes:
96                   self.zku[k][u] = self.wk_uv[k][u].sum() - 1 / np.e
97   
98       def get_pk_uv(self):
99           for k in range(0, self.diameter):
100              for u in self.nodes:
101                  for v in self.nodes:
102                      self.pk_uv[k][u][v] = self.wk_uv[k][u][v] / self.zku[k][u]
103  
104      def get_pku(self):
105          for k in range(0, self.diameter):
106              for u in self.nodes:
107                  # self.pku[k][u][0]: pk(uk, uk+1) = w(uk, uk+1) / (w(uk, uk+1) + w(uk, uk-1))
108                  # self.pku[k][u][1]: 1 - pk(uk, uk+1)
109                  self.pku[k][u][0] = self.wku[k][u][0] / (self.wku[k][u][0] + self.wku[k][u][1])
110                  self.pku[k][u][1] = 1 - self.pku[k][u][0]

112      @staticmethod
113      def alias_setup(probs):
114          ''' 
115          :param probs: 某个概率分布 
116          :return: Alias数组与Prob数组 
117          '''
118          K = len(probs)  # K为类别数目
119          Prob = np.zeros(K)  # 对应Prob数组：落在原类型的概率
120          Alias = np.zeros(K, dtype=np.int)  # 对应Alias数组：每一列第二层的类型
121  
122          # Sort the data into the outcomes with probabilities
123          # that are larger and smaller than 1/K
124          smaller = []  # 存储比1小的列
125          larger = []  # 存储比1大的列
126  
127          for kk, prob in enumerate(probs):
128              Prob[kk] = K * prob  # 概率（每个类别概率乘以K，使得总和为K）
129              if Prob[kk] < 1.0:  # 然后分为两类：大于1的和小于1的
130                  smaller.append(kk)
131              else:
132                  larger.append(kk)
133  
134          # Loop though and create little binary mixtures that appropriately allocate
135          # the larger outcomes over the overall uniform mixture.
136  
137          # 通过拼凑，将各个类别都凑为1
138          while len(smaller) > 0 and len(larger) > 0:
139              small = smaller.pop()
140              large = larger.pop()
141  
142              Alias[small] = large  # 填充Alias数组
143              Prob[large] = Prob[large] - (1.0 - Prob[small])  # 将大的分到小的上
144  
145              if Prob[large] < 1.0:
146                  smaller.append(large)
147              else:
148                  larger.append(large)
149          # print("Prob is :", Prob)
150          # print("Alias is :", Alias)
151          return Alias, Prob
152  
153      @staticmethod
154      def alias_draw(Alias, Prob):
155          ''' 
156          :param J: Alias数组 
157          :param q: Prob数组 
158          :return:一次采样结果 
159          '''
160          K = len(Alias)
161  
162          # Draw from the overall uniform mixture.
163          kk = int(np.floor(np.random.rand() * K))  # 随机取一列
164  
165          # Draw from the binary mixture, either keeping the small one, or choosing the associated larger one.
166          # 采样过程：随机取某一列k（即[1,4]的随机整数，再随机产生一个[0-1]的小数c，）
167          # 如果Prob[kk]大于c，
168          if np.random.rand() < Prob[kk]:  # 比较
169              return kk
170          else:
171              return Alias[kk]

173      def get_walks(self):
174          for u in self.nodes:
175              for i in range(0, self.num_walks):
176                  path = [str(u)]
177                  v = u
178                  k = 0
179                  # print('u = ', u)
180                  while len(path) < self.walk_length:
181                      r = np.random.random()
182                      # print('r = ', r)
183                      # before each step, the random walk first decides if it will change layers or walk on the current layer
184                      if r < 0.3:
185                          # print('select next node...')
186                          probs = self.pk_uv[k][v]
187                          # print('probs = ', probs)
188                          J, q = self.alias_setup(probs=probs)
189                          v = self.alias_draw(J, q)
190                          # print('v = ', v)
191                          path.append(str(v))
192                      else:
193                          # print('decided change layers, layer = ', k)
194                          r = np.random.random()
195                          layer_up_prob = self.pku[k][v][0]
196                          if r < layer_up_prob:
197                              k = k + 1 if k < self.diameter - 1 else k
198                          else:
199                              k = k - 1 if k > 0 else 0
200                          # print('up prob = %f, now k = %d' % (layer_up_prob, k))
201                          # print('++++++++++++++++++++++++++++++++++++++++++++++++++++++++')
202                  self.all_paths.append(path)
